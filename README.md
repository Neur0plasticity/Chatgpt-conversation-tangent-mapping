class LinkedList {

  }

  addNode(topic, subtopic) {
    const newNode = new Node(topic, subtopic);

    if (!this.head) {
      this.head = newNode;
    } else {
      let current = this.head;
      while (current.next) {
        current = current.next;
      }
      current.next = newNode;
    }

    let current = this.head;
    while (current) {
      if (current.subtopic === subtopic) {
        current.branch = new Node(topic, subtopicBranch);
        break;
      }
      current = current.next;
    }
  }
}
